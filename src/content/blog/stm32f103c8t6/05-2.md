# 1. 配置外部中断

1. 配置RCC，gpio和AFIO，EXIT本应该配置时钟唤醒，但是不需要开启时钟，NVIC是内核，不用配置
2. 配置GPIO
3. 配置AFIO选择GPIOx
4. 配置EXIT选择触发方式和响应方式
5. 配置NVIC选择合适的优先级

```c
void CounterSensor_Init(void){
	
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB,ENABLE);
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_AFIO,ENABLE);
	
	GPIO_InitTypeDef GPIO_InitStruc;
	GPIO_InitStruc.GPIO_Pin=GPIO_Pin_14;
	GPIO_InitStruc.GPIO_Mode=GPIO_Mode_IPU ;
	GPIO_InitStruc.GPIO_Speed=GPIO_Speed_50MHz ;
	
	GPIO_EXTILineConfig(GPIO_PortSourceGPIOB,GPIO_PinSource14); 
	
	EXTI_InitTypeDef EXTI_InitStruct;
	EXTI_InitStruct.EXTI_Line=EXTI_Line14;
	EXTI_InitStruct.EXTI_LineCmd=ENABLE;
	EXTI_InitStruct.EXTI_Mode=EXTI_Mode_Interrupt;
	EXTI_InitStruct.EXTI_Trigger=EXTI_Trigger_Falling;
	
	EXTI_Init(&EXTI_InitStruct);
	//整个程序只执行一次
	NVIC_PriorityGroupConfig(NVIC_PriorityGroup_2);
	
	NVIC_InitTypeDef NVIC_InitStruct;
	NVIC_InitStruct.NVIC_IRQChannel=EXTI15_10_IRQn;//因为选择的line是14索引对应的通道是15-10
	NVIC_InitStruct.NVIC_IRQChannelCmd=ENABLE;
	NVIC_InitStruct.NVIC_IRQChannelPreemptionPriority=1;
	NVIC_InitStruct.NVIC_IRQChannelSubPriority=1;
	NVIC_Init(&NVIC_InitStruct);
}
```

6. 编写中断函数：
   1. 中断函数名在start/下的启动文件里查看向量表IQRCommander
   2. 中断函数和main函数都是编写定义，不编写执行逻辑，逻辑是由stm32系统自动执行
   3. 在中断内操作变量值和标志位，避免外设中断没有中断保护
